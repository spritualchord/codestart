#include <iostream>
#include <stack>

using namespace std;

stack<int> addNum(stack<int> s1, stack<int> s2) {
    stack<int> result;
    int carry = 0;

    while (!s1.empty() || !s2.empty() || carry) {
        int sum = carry;
        if (!s1.empty()) {
            sum += s1.top();
            s1.pop();
        }
        if (!s2.empty()) {
            sum += s2.top();
            s2.pop();
        }

        carry = sum / 10;
        sum = sum % 10;
        result.push(sum);
    }

    return result;
}

int main() {
    stack<int> num1, num2, sum;

    num1.push(5);
    num1.push(4);
    num1.push(3);
    num1.push(2);

    num2.push(1);
    num2.push(2);
    num2.push(3);
    num2.push(4);

    sum = addNum(num1, num2);

    cout << endl << "Sum of 2 Large Numbers : ";
    while (!sum.empty()) {
        cout << sum.top();
        sum.pop();
    }
    cout << endl;

    return 0;
}
